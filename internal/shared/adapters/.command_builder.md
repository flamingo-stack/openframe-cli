A fluent API wrapper around the Cobra CLI framework that provides builder patterns for command construction and utilities for flag extraction and validation.

## Key Components

- **CommandBuilder** - Fluent interface for building cobra commands with method chaining
- **FlagExtractor** - Utility for safely extracting and validating command flags
- **ValidationResult** - Structure for collecting and managing validation errors
- **NewCommandBuilder()** - Factory function for creating command builders
- **NewFlagExtractor()** - Factory function for creating flag extractors

## Usage Example

```go
// Build a command with fluent interface
cmd := NewCommandBuilder("serve", "Start the server").
    Long("Start the HTTP server on the specified port").
    Aliases([]string{"s", "start"}).
    AddStringFlag("port", "p", "8080", "Server port").
    AddBoolFlag("verbose", "v", false, "Enable verbose logging").
    Args(cobra.NoArgs).
    RunE(func(cmd *cobra.Command, args []string) error {
        // Extract flags safely
        extractor := NewFlagExtractor(cmd)
        
        port, err := extractor.GetString("port")
        if err != nil {
            return err
        }
        
        verbose, err := extractor.GetBool("verbose")
        if err != nil {
            return err
        }
        
        // Use validation result
        result := NewValidationResult()
        if port == "" {
            result.AddError(fmt.Errorf("port cannot be empty"))
        }
        
        if result.HasErrors() {
            return result.GetFirstError()
        }
        
        // Start server logic here
        return nil
    }).
    Build()
```
