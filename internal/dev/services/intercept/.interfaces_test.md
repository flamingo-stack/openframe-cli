Test file providing mock implementations and comprehensive unit tests for Kubernetes client interfaces and data structures. Contains mock objects for testing Kubernetes and service operations without actual cluster dependencies.

## Key Components

- **MockKubernetesClient**: Mock implementation of KubernetesClient interface for testing namespace operations
- **MockServiceClient**: Mock implementation of ServiceClient interface for testing service operations  
- **ServicePort struct tests**: Validates port configuration structure with name, port, target port, and protocol
- **ServiceInfo struct tests**: Validates service metadata including name, namespace, type, and port collections
- **Interface compliance tests**: Ensures mock implementations satisfy the actual interfaces

## Usage Example

```go
// Testing namespace operations
func TestNamespaceValidation(t *testing.T) {
    ctx := context.Background()
    mockClient := new(MockKubernetesClient)
    
    mockClient.On("ValidateNamespace", ctx, "production").Return(nil)
    
    err := mockClient.ValidateNamespace(ctx, "production")
    assert.NoError(t, err)
    mockClient.AssertExpectations(t)
}

// Testing service retrieval
func TestServiceRetrieval(t *testing.T) {
    ctx := context.Background()
    mockClient := new(MockServiceClient)
    
    expectedService := &ServiceInfo{
        Name:      "api-service",
        Namespace: "default",
        Type:      "ClusterIP",
        Ports: []ServicePort{
            {Name: "http", Port: 8080, TargetPort: "8080", Protocol: "TCP"},
        },
    }
    
    mockClient.On("GetService", ctx, "default", "api-service").Return(expectedService, nil)
    
    service, err := mockClient.GetService(ctx, "default", "api-service")
    assert.NoError(t, err)
    assert.Equal(t, "api-service", service.Name)
}
```
