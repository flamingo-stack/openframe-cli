Defines core interfaces and data structures for Kubernetes service interception and management operations.

## Key Components

**Data Structures:**
- `ServicePort` - Configuration for individual service ports with name, port number, target port, and protocol
- `ServiceInfo` - Complete service metadata including name, namespace, type, and port configurations

**Interfaces:**
- `KubernetesClient` - Namespace operations including listing and validation
- `ServiceClient` - Service operations for retrieval and validation within namespaces

## Usage Example

```go
// Implementing a service client
type myServiceClient struct {
    // client implementation
}

func (c *myServiceClient) GetServices(ctx context.Context, namespace string) ([]ServiceInfo, error) {
    // Retrieve services from Kubernetes API
    services := []ServiceInfo{
        {
            Name:      "web-service",
            Namespace: namespace,
            Type:      "ClusterIP",
            Ports: []ServicePort{
                {
                    Name:       "http",
                    Port:       80,
                    TargetPort: "8080",
                    Protocol:   "TCP",
                },
            },
        },
    }
    return services, nil
}

// Using the interfaces
func listNamespaceServices(k8sClient KubernetesClient, svcClient ServiceClient) error {
    ctx := context.Background()
    
    namespaces, err := k8sClient.GetNamespaces(ctx)
    if err != nil {
        return err
    }
    
    for _, ns := range namespaces {
        services, err := svcClient.GetServices(ctx, ns)
        if err != nil {
            continue
        }
        // Process services...
    }
    return nil
}
```
