Defines interfaces and data structures for Kubernetes service interception operations, providing abstractions for namespace and service management within the intercept package.

## Key Components

**Data Structures:**
- `ServicePort` - Configuration for a single service port with name, port number, target port, and protocol
- `ServiceInfo` - Complete service metadata including name, namespace, type, and port configurations

**Interfaces:**
- `KubernetesClient` - Namespace operations including listing and validation
- `ServiceClient` - Service operations for retrieval and validation within namespaces

## Usage Example

```go
// Implementing the ServiceClient interface
type myServiceClient struct {
    // client implementation
}

func (c *myServiceClient) GetServices(ctx context.Context, namespace string) ([]ServiceInfo, error) {
    // Implementation to fetch services from Kubernetes
    return []ServiceInfo{
        {
            Name:      "web-service",
            Namespace: "default",
            Type:      "ClusterIP",
            Ports: []ServicePort{
                {
                    Name:       "http",
                    Port:       80,
                    TargetPort: "8080",
                    Protocol:   "TCP",
                },
            },
        },
    }, nil
}

// Using the interfaces
func interceptService(k8sClient KubernetesClient, svcClient ServiceClient) error {

    // Validate namespace exists

    }
    
    // Get service information
    service, err := svcClient.GetService(ctx, "default", "web-service")
    if err != nil {
        return err
    }
    
    // Process service ports for interception
    for _, port := range service.Ports {
        // Setup interception logic
    }
    
    return nil
}
```
