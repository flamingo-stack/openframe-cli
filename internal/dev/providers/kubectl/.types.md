Defines JSON structures for parsing Kubernetes service information from kubectl command output. These types enable type-safe unmarshaling of service data retrieved through kubectl JSON responses.

## Key Components

- **serviceJSON**: Represents a single Kubernetes service with metadata (name, namespace) and specification details (type, ports configuration)
- **serviceListJSON**: Container structure for multiple services, wrapping an array of serviceJSON items

## Usage Example

```go
import (
    "encoding/json"
    "fmt"
)

// Parse a single service from kubectl get service output
func parseService(jsonData []byte) (*serviceJSON, error) {
    var service serviceJSON
    err := json.Unmarshal(jsonData, &service)
    if err != nil {
        return nil, err
    }
    return &service, nil
}

// Parse multiple services from kubectl get services output
func parseServices(jsonData []byte) (*serviceListJSON, error) {
    var services serviceListJSON
    err := json.Unmarshal(jsonData, &services)
    if err != nil {
        return nil, err
    }
    
    fmt.Printf("Found %d services\n", len(services.Items))
    for _, svc := range services.Items {
        fmt.Printf("Service: %s/%s (Type: %s)\n", 
            svc.Metadata.Namespace, 
            svc.Metadata.Name, 
            svc.Spec.Type)
    }
    return &services, nil
}
```
