This file defines JSON structure types for parsing kubectl command output related to Kubernetes services.

## Key Components

- **`serviceJSON`** - Represents a single Kubernetes service with metadata (name, namespace) and spec (type, ports configuration)
- **`serviceListJSON`** - Represents a collection of services returned by kubectl list operations

## Usage Example

```go
package main

import (
    "encoding/json"
    "fmt"
)

func parseKubectlServiceOutput(jsonData []byte) error {
    var serviceList serviceListJSON
    if err := json.Unmarshal(jsonData, &serviceList); err != nil {
        return err
    }
    
    for _, service := range serviceList.Items {
        fmt.Printf("Service: %s/%s (Type: %s)\n", 
            service.Metadata.Namespace, 
            service.Metadata.Name, 
            service.Spec.Type)
        
        for _, port := range service.Spec.Ports {
            fmt.Printf("  Port: %d -> %v (%s)\n", 
                port.Port, port.TargetPort, port.Protocol)
        }
    }
    return nil
}
```

These types are typically used to unmarshal JSON output from commands like `kubectl get services -o json`.
