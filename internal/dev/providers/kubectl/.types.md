This file defines JSON parsing structures for kubectl service-related commands, enabling type-safe deserialization of Kubernetes service data.

## Key Components

- **serviceJSON**: Represents a single Kubernetes service with metadata (name, namespace) and spec (type, ports configuration)
- **serviceListJSON**: Container structure for parsing lists of services from kubectl output

## Usage Example

```go
package main

import (
    "encoding/json"
    "fmt"
)

func parseServiceOutput(jsonData []byte) error {
    var serviceList serviceListJSON
    if err := json.Unmarshal(jsonData, &serviceList); err != nil {
        return err
    }
    
    for _, service := range serviceList.Items {
        fmt.Printf("Service: %s/%s (Type: %s)\n", 
            service.Metadata.Namespace, 
            service.Metadata.Name, 
            service.Spec.Type)
        
        for _, port := range service.Spec.Ports {
            fmt.Printf("  Port: %d -> %v (%s)\n", 
                port.Port, port.TargetPort, port.Protocol)
        }
    }
    return nil
}
```

These structures are typically used with `kubectl get services -o json` output to programmatically process Kubernetes service information.
