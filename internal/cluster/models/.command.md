This file defines the core interfaces for implementing cluster commands in the CLI application, providing a standardized contract for command structure, execution, and flag handling.

## Key Components

- **CommandFlags** - Interface for accessing global command flags
- **CommandExecutor** - Interface for accessing the command execution context
- **ClusterCommand** - Main interface that all cluster commands must implement, defining methods for:
  - Getting the cobra command instance
  - Executing command logic
  - Validating flags
  - Setting up command-specific flags

## Usage Example

```go
// Implementing a cluster command
type MyClusterCommand struct{}

func (c *MyClusterCommand) GetCommand(flags CommandFlags) *cobra.Command {
    cmd := &cobra.Command{
        Use:   "my-command",
        Short: "Description of my command",
    }
    c.SetupFlags(cmd, flags)
    return cmd
}

func (c *MyClusterCommand) Execute(cmd *cobra.Command, args []string, flags CommandFlags, executor CommandExecutor) error {
    // Validate flags first
    if err := c.ValidateFlags(flags); err != nil {
        return err
    }
    
    // Execute command logic using the executor
    return executor.GetExecutor().Run("some-command", args...)
}

    return nil // Implement validation logic
}

    // Add command-specific flags
}
```
