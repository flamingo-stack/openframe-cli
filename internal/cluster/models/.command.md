Defines the core interfaces for implementing structured CLI commands with standardized flag handling and execution patterns.

## Key Components

- **`CommandFlags`** - Interface for accessing global command flags across all commands
- **`CommandExecutor`** - Interface providing access to the underlying command execution engine
- **`ClusterCommand`** - Main interface that all cluster-related commands must implement, defining the complete command lifecycle

## Usage Example

```go
type MyClusterCommand struct{}

func (c *MyClusterCommand) GetCommand(flags CommandFlags) *cobra.Command {
    cmd := &cobra.Command{
        Use:   "deploy",
        Short: "Deploy cluster resources",
    }
    c.SetupFlags(cmd, flags)
    return cmd
}

func (c *MyClusterCommand) Execute(cmd *cobra.Command, args []string, flags CommandFlags, executor CommandExecutor) error {
    // Validate input
    if err := c.ValidateFlags(flags); err != nil {
        return err
    }
    
    // Execute command logic using the provided executor
    return executor.GetExecutor().RunCommand("kubectl", "apply", "-f", "config.yaml")
}

    return nil // Add validation logic
}

func (c *MyClusterCommand) SetupFlags(cmd *cobra.Command, flags CommandFlags) {
    // Add command-specific flags
}
```
