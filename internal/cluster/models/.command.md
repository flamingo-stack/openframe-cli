This file defines the core interfaces for implementing CLI commands in the openframe-cli system, establishing a contract for command structure, execution, and flag handling.

## Key Components

- **CommandFlags**: Interface for accessing global command flags
- **CommandExecutor**: Interface providing access to command execution context
- **ClusterCommand**: Main interface defining the complete command lifecycle including setup, validation, and execution

## Usage Example

```go
type MyCommand struct{}

func (c *MyCommand) GetCommand(flags CommandFlags) *cobra.Command {
    cmd := &cobra.Command{
        Use:   "mycommand",
        Short: "Example command",
    }
    c.SetupFlags(cmd, flags)
    return cmd
}

func (c *MyCommand) Execute(cmd *cobra.Command, args []string, flags CommandFlags, executor CommandExecutor) error {
    // Command implementation
    return executor.GetExecutor().Run("some-command", args...)
}

func (c *MyCommand) ValidateFlags(flags CommandFlags) error {
    // Validation logic
    return nil
}

func (c *MyCommand) SetupFlags(cmd *cobra.Command, flags CommandFlags) {
    // Add command-specific flags
    cmd.Flags().StringP("option", "o", "", "Example option")
}
```
