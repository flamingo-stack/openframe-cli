This file defines the core interfaces for the command pattern used throughout the OpenFrame CLI, providing a standardized structure for implementing Cobra commands with shared flag handling and execution context.

## Key Components

- **`CommandFlags`** - Interface for accessing global command flags across all commands
- **`CommandExecutor`** - Interface providing access to the command execution engine for running system operations
- **`ClusterCommand`** - Main interface that all cluster commands must implement, defining the complete command lifecycle

## Usage Example

```go
type MyClusterCommand struct {
    name string
}

func (c *MyClusterCommand) GetCommand(flags CommandFlags) *cobra.Command {
    cmd := &cobra.Command{
        Use:   "mycommand",
        Short: "Description of my command",
    }
    c.SetupFlags(cmd, flags)
    return cmd
}

func (c *MyClusterCommand) Execute(cmd *cobra.Command, args []string, flags CommandFlags, executor CommandExecutor) error {
    // Validate input
    if err := c.ValidateFlags(flags); err != nil {
        return err
    }
    
    // Use executor to run operations
    return executor.GetExecutor().Run("kubectl", "get", "pods")
}

}
```
