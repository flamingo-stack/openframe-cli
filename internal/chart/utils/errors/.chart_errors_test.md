This file contains comprehensive unit tests for a chart error handling system, covering custom error types and utility functions for Kubernetes/Helm operations.

## Key Components

- **ChartError Tests**: Core error type with operation, component, and recoverability tracking
- **InstallationError Tests**: Specialized error for chart installation phases with troubleshooting steps
- **ValidationError Tests**: Field validation errors with constraint checking
- **ConfigurationError Tests**: Configuration file parsing errors with missing key tracking
- **Utility Function Tests**: Error classification, wrapping, and combining helpers

## Usage Example

```go
// Test basic chart error creation
func TestNewChartError(t *testing.T) {
    cause := errors.New("test error")
    chartErr := NewChartError("installation", "ArgoCD", cause)
    
    assert.Equal(t, "installation", chartErr.Operation)
    assert.Equal(t, "ArgoCD", chartErr.Component)
    assert.False(t, chartErr.Recoverable)
}

// Test recoverable error with retry logic
func TestRecoverableError(t *testing.T) {
    retryAfter := 30 * time.Second
    chartErr := NewRecoverableChartError("installation", "Helm", cause, retryAfter)
    
    assert.True(t, chartErr.IsRecoverable())
    assert.Equal(t, retryAfter, chartErr.GetRetryAfter())
}

// Test error utility functions
func TestErrorHelpers(t *testing.T) {
    assert.True(t, IsTimeout(timeoutErr))
    assert.True(t, IsRecoverable(recoverableErr))
    assert.Equal(t, 15*time.Second, GetRetryDelay(recoverableErr))
}
```
